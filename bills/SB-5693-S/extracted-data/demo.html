Query Examples Using Lodash

Example 1: Filter Appropriations by Agency

const commerceApprops = _.filter(billData.appropriations, { agency: 'DEPARTMENT OF COMMERCE' }); console.log(commerceApprops);

Example 2: Calculate Total for FY 2023

const fy2023Apprps = _.filter(billData.appropriations, { fiscalYear: 'FY 2023' }); const total = _.sumBy(fy2023Apprps, item => parseMoneyValue(item.newAmount)); console.log('FY 2023 Total: $' + total.toLocaleString());

Example 3: Find All Vetoed Sections

const vetoedSections = _.map(billData.vetoes, veto => ({ section: veto.sectionNumber, type: veto.vetoType, lineVetoes: veto.lineVetoes.length })); console.log(vetoedSections);

Example 4: Group Appropriations by Agency

const byAgency = _.groupBy(billData.appropriations, 'agency'); const agencyTotals = _.mapValues(byAgency, items => _.sumBy(items, i => parseMoneyValue(i.newAmount))); console.log(agencyTotals);

Example 5: Find Large Programs (>$1M)

const largePrograms = _.filter(billData.programs, program => { return program.amounts.some(amt => parseMoneyValue(amt) > 1000000 ); }); console.log(largePrograms.length + ' programs over $1M');
